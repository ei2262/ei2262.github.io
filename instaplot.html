data(rest_inspec)

boro_rest = rest_inspec %>% 
 select(boro, violation_description, action, score, dba, cuisine_description, inspection_type, critical_flag) %>% 
  mutate(score=cut(rest_inspec$score, breaks=c(0,13,27,Inf), labels=c("A","B","C"))) %>% 
  drop_na(score)

boro_rest %>%
  count(boro) %>% 
  mutate(boro = fct_reorder(boro, n)) %>% 
  plot_ly(x = ~boro, y = ~n, color = ~boro, type = "bar", colors = "viridis")
